GCD PROGRAM:
a=int(input("Enter the first number: "))
b=int(input("Enter the second number: "))
if a==0 or b==0:
    print("Value must be non zero")
else:
    if a<b:
        result=a
    else:
        result=b
    while result:
        if a%result ==0 and b%result==0:
            break
        result=result-1
print("gcd: ",result)
